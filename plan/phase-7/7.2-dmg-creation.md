# 7.2 DMG Creation

**Objective:** Create professional distributable DMG with custom background and layout.

## Implementation

### Step 1: DMG Creation Script

Create `create-dmg.sh`:

```bash
#!/bin/bash

set -e

APP_NAME="FocusDragon"
VERSION="1.0.0"
DMG_NAME="${APP_NAME}-${VERSION}"
SOURCE_APP="build/Release/${APP_NAME}.app"
DMG_DIR="build/dmg"
DMG_PATH="build/${DMG_NAME}.dmg"

# Clean
rm -rf "$DMG_DIR"
rm -f "$DMG_PATH"

# Create DMG directory
mkdir -p "$DMG_DIR"

# Copy app
cp -R "$SOURCE_APP" "$DMG_DIR/"

# Create Applications symlink
ln -s /Applications "$DMG_DIR/Applications"

# Create background folder
mkdir -p "$DMG_DIR/.background"
cp "resources/dmg-background.png" "$DMG_DIR/.background/"

# Create temporary DMG
hdiutil create -srcfolder "$DMG_DIR" -volname "$APP_NAME" -fs HFS+ \\
    -fsargs "-c c=64,a=16,e=16" -format UDRW -size 200m "build/temp.dmg"

# Mount it
MOUNT_DIR=$(hdiutil attach -readwrite -noverify -noautoopen "build/temp.dmg" | grep /Volumes | awk '{print $3}')

# Set custom icon
cp resources/VolumeIcon.icns "$MOUNT_DIR/.VolumeIcon.icns"
SetFile -c icnC "$MOUNT_DIR/.VolumeIcon.icns"

# Configure appearance
echo '
   tell application "Finder"
     tell disk "'$APP_NAME'"
           open
           set current view of container window to icon view
           set toolbar visible of container window to false
           set statusbar visible of container window to false
           set the bounds of container window to {100, 100, 700, 500}
           set viewOptions to the icon view options of container window
           set arrangement of viewOptions to not arranged
           set icon size of viewOptions to 72
           set background picture of viewOptions to file ".background:dmg-background.png"
           set position of item "'$APP_NAME'.app" of container window to {150, 200}
           set position of item "Applications" of container window to {450, 200}
           close
           open
           update without registering applications
           delay 2
     end tell
   end tell
' | osascript

# Unmount
hdiutil detach "$MOUNT_DIR"

# Convert to compressed read-only
hdiutil convert "build/temp.dmg" -format UDZO -imagekey zlib-level=9 -o "$DMG_PATH"

# Clean up
rm -rf "$DMG_DIR"
rm "build/temp.dmg"

echo "DMG created: $DMG_PATH"

# Code sign DMG
codesign -s "Developer ID Application: YOUR_NAME (TEAM_ID)" "$DMG_PATH"

echo "DMG signed and ready for distribution!"
```
